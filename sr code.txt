import threading
import os

def run_receiver():
    os.system("python receive.py")

def run_sender():
    os.system("python send.py 10.0.2.2 1002")

receiver_thread = threading.Thread(target=run_receiver)
sender_thread = threading.Thread(target=run_sender)

receiver_thread.start()
sender_thread.start()

receiver_thread.join()
sender_thread.join()
